016    public class FileWraper {
017        public InputStream inputStream;
018        public String fileName;
019        public String contentType;
020    
021        public FileWraper(InputStream inputStream, String fileName, String contentType) {
022            this.inputStream = inputStream;
023            this.fileName = fileName;
024            this.contentType = contentType;
025        }
026    
027        public String getFileName() {
028            return (null != fileName) ? fileName : "nofilename";
