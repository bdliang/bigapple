006    package com.winupon.andframe.bigapple.utils.textviewhtml.helper;
007    
008    import android.content.Context;
009    import android.graphics.Bitmap;
010    import android.graphics.drawable.BitmapDrawable;
011    import android.graphics.drawable.Drawable;
012    import android.text.Spannable;
013    import android.text.SpannableString;
014    import android.text.style.ImageSpan;
015    
016    /**
017     * 可以处理图标和文字添加到EditText中，使用span包里的API代替
018     * 
019     * @author xuan
020     * @version $Revision: 1.0 $, $Date: 2013-4-16 下午7:50:24 $
021     */
022    @Deprecated
023    public abstract class ImageSpanUtils {
024    
025        /**
026         * 图片代替文字显示
027         * 
028         * @param context
029         * @param text
030         * @param resId
031         * @return
032         */
033        public static SpannableString getSpannableStringByTextReplaceBitmap(Context context, String text, int resId) {
034            Drawable drawable = context.getResources().getDrawable(resId);
035            BitmapDrawable bd = (BitmapDrawable) drawable;
036            return getSpannableStringByTextReplaceBitmap(context, text, bd.getBitmap());
037        }
038    
039        /**
040         * 图片代替文字显示
041         * 
042         * @param context
043         * @param text
044         * @param bitmap
045         * @return
046         */
047        public static SpannableString getSpannableStringByTextReplaceBitmap(Context context, String text, Bitmap bitmap) {
048            ImageSpan imageSpan = new ImageSpan(context, bitmap);
049            SpannableString spannableString = new SpannableString(text);
050            spannableString.setSpan(imageSpan, 0, text.length(), Spannable.SPAN_EXCLUSIVE_EXCLUSIVE);
051            return spannableString;
052        }
053    
054    }
