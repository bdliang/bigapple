001    package com.winupon.andframe.bigapple.io;
002    
003    import java.nio.charset.Charset;
004    
005    /**
006     * 常用编码工具
007     * 
008     * @author xuan
009     * @version $Revision: 1.0 $, $Date: 2013-9-4 下午7:32:50 $
010     */
011    public class Charsets {
012    
013        /**
014         * 获取字符编码对象
015         * 
016         * @param charset
017         * @return
018         */
019        public static Charset toCharset(Charset charset) {
020            return charset == null ? Charset.defaultCharset() : charset;
021        }
022    
023        /**
024         * 获取字符编码对象
025         * 
026         * @param charset
027         * @return
028         */
029        public static Charset toCharset(String charset) {
030            return charset == null ? Charset.defaultCharset() : Charset.forName(charset);
031        }
032    
033        public static final String UTF8 = "UTF-8";
034        public static final String ISO88591 = "ISO-8859-1";
035        public static final String USASCII = "US-ASCII";
036        public static final String UTF16 = "UTF-16";
037        public static final String UTF16BE = "UTF-16BE";
038        public static final String UTF16LE = "UTF-16LE";
039    
040        public static final Charset UTF_8 = Charset.forName(UTF8);
041        public static final Charset ISO_8859_1 = Charset.forName(ISO88591);
042        public static final Charset US_ASCII = Charset.forName(USASCII);
043        public static final Charset UTF_16 = Charset.forName(UTF16);
044        public static final Charset UTF_16BE = Charset.forName(UTF16BE);
045        public static final Charset UTF_16LE = Charset.forName(UTF16LE);
046    
047    }
