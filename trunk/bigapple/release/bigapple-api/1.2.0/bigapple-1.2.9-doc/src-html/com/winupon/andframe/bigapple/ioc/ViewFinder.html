006    package com.winupon.andframe.bigapple.ioc;
007    
008    import android.app.Activity;
009    import android.view.View;
010    
011    /**
012     * 注入View的寻找器
013     * 
014     * @author xuan
015     * @version $Revision: 1.0 $, $Date: 2013-10-28 下午5:37:24 $
016     */
017    public class ViewFinder {
018        private View view;
019        private Activity activity;
020    
021        public ViewFinder(View view) {
022            this.view = view;
023        }
024    
025        public ViewFinder(Activity activity) {
026            this.activity = activity;
027        }
028    
029        /**
030         * 从寻找器中找到指定的View
031         * 
032         * @param id
033         * @return
034         */
035        public View findViewById(int id) {
036            return null == activity ? view.findViewById(id) : activity.findViewById(id);
037        }
038    
039    }
