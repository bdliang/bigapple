157        // Ë∑üË∏™Âà∞Ë∞ÉÁî®ËØ•Êó•ÂøóÁöÑÊñπÊ≥?    private static StackTraceElement getCallerMethodName() {
158            StackTraceElement[] stacks = Thread.currentThread().getStackTrace();

162        // ËßÑËåÉTAGÊ†ºÂºèÔºöÁ±ªÂêç[ÊñπÊ≥ïÂêçÔºå Ë∞ÉÁî®Ë°åÊï∞]
163        private static String generateTag(StackTraceElement caller) {
164            String tag = "%s[%s, %d]";
165            String callerClazzName = caller.getClassName();
166            callerClazzName = callerClazzName.substring(callerClazzName.lastIndexOf(".") + 1);
167            tag = String.format(tag, callerClazzName, caller.getMethodName(), caller.getLineNumber());
168            return tag;
169        }
