019    public class UrlEncodedFormEntity implements RequestEntiy {
020            private String encode = "utf-8";
021            private final RequestParams requestParams;
022    
023            public UrlEncodedFormEntity(RequestParams requestParams) {
024                    this.requestParams = requestParams;
025            }
026    
027            public UrlEncodedFormEntity(RequestParams requestParams, String encode) {
028                    this.requestParams = requestParams;
029                    this.encode = encode;
030            }
031    
032            /**
033             * ÊääÂÜÖÂÆπÂ≠òÂÖ•ËæìÂá∫ÊµÅ‰∏?
034             * 
035             * @param outstream
036             * @throws IOException
037             */
038            @Override
039            public void writeTo(final OutputStream outstream) throws IOException {
040                    Map<String, String> paramMap = null;
041                    if (null != requestParams) {
042                            paramMap = requestParams.getParamMap();
043                    }
044    
045                    if (null != paramMap && !paramMap.isEmpty()
046                                    && !TextUtils.isEmpty(encode)) {
047    
048                            StringBuilder sb = new StringBuilder();
049                            for (Map.Entry<String, String> e : paramMap.entrySet()) {
050                                    sb.append(URLEncoder.encode(e.getKey(), encode));
051                                    sb.append("=");
052                                    sb.append(URLEncoder.encode(e.getValue(), encode));
053                                    sb.append("&");
054                            }
055    
056                            DataOutputStream out = new DataOutputStream(outstream);
057                            out.writeBytes(sb.toString());
058                            out.flush();
059                            IOUtils.closeQuietly(out);
060                    }
061            }
062    
063    }
