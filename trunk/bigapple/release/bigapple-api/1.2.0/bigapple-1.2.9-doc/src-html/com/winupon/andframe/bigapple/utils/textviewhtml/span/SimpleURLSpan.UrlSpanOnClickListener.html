017    public class SimpleURLSpan extends URLSpan {
018        private final UrlSpanOnClickListener urlSpanOnClickListener;
019    
020        public SimpleURLSpan(String url, UrlSpanOnClickListener urlSpanOnClickListener) {
021            super(url);
022            this.urlSpanOnClickListener = urlSpanOnClickListener;
023        }
024    
025        @Override
026        public void onClick(View widget) {
027            if (null != urlSpanOnClickListener) {
028                urlSpanOnClickListener.onClick(widget, getURL());
029            }
030            else {
031                super.onClick(widget);
032            }
033        }
034
