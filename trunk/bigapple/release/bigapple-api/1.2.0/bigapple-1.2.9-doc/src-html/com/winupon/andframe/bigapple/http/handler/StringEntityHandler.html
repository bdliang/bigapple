015    public class StringEntityHandler {
016        private static int BUFFER_SIZE = 1024;
017    
018        /**
019         * Â§ÑÁêÜ
020         * 
021         * @param entity
022         *            Â§ÑÁêÜÂÆû‰Ωì
023         * @param callback
024         *            Â§ÑÁêÜÊó∂ÁöÑÂõûË∞É

027         * 
028         * @return
029         * @throws IOException
030         */
031        public Object handleEntity(HttpEntity entity, EntityCallBack callback, String charset) throws IOException {
032            if (entity == null) {
033                return null;
034            }
035    
036            ByteArrayOutputStream outStream = new ByteArrayOutputStream();
037            byte[] buffer = new byte[BUFFER_SIZE];
038    
039            long count = entity.getContentLength();
040            long curCount = 0;
041            int len = -1;
042            InputStream is = entity.getContent();
043            while ((len = is.read(buffer)) != -1) {
044                outStream.write(buffer, 0, len);
045                curCount += len;
046                if (callback != null) {
047                    callback.callBack(count, curCount, false);
048                }
049            }
050    
051            if (callback != null) {
052                callback.callBack(count, curCount, true);
053            }
054    
055            byte[] data = outStream.toByteArray();
056            outStream.close();
057            is.close();
058            return new String(data, charset);
059        }
060
