001    package com.winupon.andframe.bigapple.http2.request;
002    
003    import java.net.MalformedURLException;
004    
005    import com.winupon.andframe.bigapple.http2.request.HttpRequest;
006    import com.winupon.andframe.bigapple.http2.request.entity.RequestEntiy;
007    
008    /**
009     * GET请求方法
010     * 
011     * @author xuan
012     */
013    public class HttpPost extends HttpRequest {
014            private RequestEntiy requestEntiy;
015    
016            public HttpPost(String urlStr) throws MalformedURLException {
017                    super(urlStr);
018            }
019    
020            public HttpPost(String urlStr, RequestEntiy requestEntiy)
021                            throws MalformedURLException {
022                    super(urlStr);
023                    this.requestEntiy = requestEntiy;
024            }
025    
026            public RequestEntiy getRequestEntiy() {
027                    return requestEntiy;
028            }
029    
030            public void setRequestEntiy(RequestEntiy requestEntiy) {
031                    this.requestEntiy = requestEntiy;
032            }
033    
034    }
